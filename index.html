        if (isEmpty())
           return false;
        else {
           ListNode<NODETYPE> *tempPtr = firstNode;
           if (firstNode == lastNode)
              firstNode = lastNode = NULL;
           else
              firstNode = firstNode -> nextNode;
           delete tempPtr;
           size --;
 return true;
        }

     ListNode<NODETYPE> *newPtr = getNewNode(newData);
     if (isEmpty()) {
          firstNode = lastNode = newPtr;
     }
     else {
          lastNode -> nextNode = newPtr;
          lastNode = newPtr;
     }
     size++;
}
/* function removeFromTail:  remove the element pointed by the last node  of the LinkedList */
template <typename NODETYPE>
bool LinkedList<NODETYPE> :: removeFromTail() {

        if (isEmpty())
            return false;
        else {
           ListNode<NODETYPE> *tempPtr = firstNode;

           if (firstNode  == lastNode)
              firstNode = lastNode = NULL;
           else  {
              while (tempPtr -> nextNode != lastNode) {
                 tempPtr = tempPtr -> nextNode;
              }
             lastNode = tempPtr;
              tempPtr = tempPtr -> nextNode;
              lastNode -> nextNode = NULL;
          }
          delete tempPtr;
          size --;
          return true;
        }
}

       data = newData;
       leftNode = rightNode = NULL;

}

// Function insert
template<typename NODETYPE>
void Node<NODETYPE>:: insert(NODETYPE &newData) {

        if (newData < data) {
           if (leftNode == NULL)
              leftNode = new Node<NODETYPE>(newData);
           else
               leftNode -> insert(newData);
        }
        else if (newData > data) {
           if (rightNode == NULL)
              rightNode = new Node<NODETYPE> (newData);
           else rightNode -> insert(newData);
        }
}
